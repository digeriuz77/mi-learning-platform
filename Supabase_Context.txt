##create table public.conversation_analyses (
  id uuid not null default gen_random_uuid (),
  session_id text not null,
  conversation_id text null,
  user_id uuid null,
  persona_id text null,
  persona_name text null,
  overall_score numeric(3, 2) null,
  foundational_trust_safety numeric(3, 2) null,
  empathic_partnership_autonomy numeric(3, 2) null,
  empowerment_clarity numeric(3, 2) null,
  mi_spirit_score numeric(3, 2) null,
  partnership_demonstrated boolean null default false,
  acceptance_demonstrated boolean null default false,
  compassion_demonstrated boolean null default false,
  evocation_demonstrated boolean null default false,
  techniques_count jsonb null default '{}'::jsonb,
  techniques_used jsonb null default '[]'::jsonb,
  strengths jsonb null default '[]'::jsonb,
  areas_for_improvement jsonb null default '[]'::jsonb,
  client_movement text null,
  change_talk_evoked boolean null default false,
  transcript_summary text null,
  summary text null,
  key_moments jsonb null default '[]'::jsonb,
  suggestions_for_next_time jsonb null default '[]'::jsonb,
  transcript jsonb null default '[]'::jsonb,
  total_turns integer null default 0,
  created_at timestamp with time zone null default now(),
  metadata jsonb null default '{}'::jsonb,
  constraint conversation_analyses_pkey primary key (id),
  constraint conversation_analyses_user_id_fkey foreign KEY (user_id) references auth.users (id),
  constraint conversation_analyses_empowerment_clarity_check check (
    (
      (empowerment_clarity >= (1)::numeric)
      and (empowerment_clarity <= (5)::numeric)
    )
  ),
  constraint conversation_analyses_foundational_trust_safety_check check (
    (
      (foundational_trust_safety >= (1)::numeric)
      and (foundational_trust_safety <= (5)::numeric)
    )
  ),
  constraint conversation_analyses_client_movement_check check (
    (
      client_movement = any (
        array[
          'toward_change'::text,
          'stable'::text,
          'away_from_change'::text
        ]
      )
    )
  ),
  constraint conversation_analyses_overall_score_check check (
    (
      (overall_score >= (1)::numeric)
      and (overall_score <= (5)::numeric)
    )
  ),
  constraint conversation_analyses_mi_spirit_score_check check (
    (
      (mi_spirit_score >= (1)::numeric)
      and (mi_spirit_score <= (5)::numeric)
    )
  ),
  constraint conversation_analyses_empathic_partnership_autonomy_check check (
    (
      (empathic_partnership_autonomy >= (1)::numeric)
      and (empathic_partnership_autonomy <= (5)::numeric)
    )
  )
) TABLESPACE pg_default;

create index IF not exists idx_conversation_analyses_user_id on public.conversation_analyses using btree (user_id) TABLESPACE pg_default;

create index IF not exists idx_conversation_analyses_session_id on public.conversation_analyses using btree (session_id) TABLESPACE pg_default;

create index IF not exists idx_conversation_analyses_persona_id on public.conversation_analyses using btree (persona_id) TABLESPACE pg_default;

create index IF not exists idx_conversation_analyses_created_at on public.conversation_analyses using btree (created_at) TABLESPACE pg_default;

create trigger update_user_analytics_on_analysis
after INSERT on conversation_analyses for EACH row
execute FUNCTION trigger_update_user_analytics ();

##create table public.user_progress (
  id uuid not null default gen_random_uuid (),
  user_id uuid not null,
  module_id uuid not null,
  status character varying(20) null default 'not_started'::character varying,
  current_node_id character varying(50) null default 'node_1'::character varying,
  nodes_completed text[] null default '{}'::text[],
  points_earned integer null default 0,
  completion_score integer null default 0,
  techniques_demonstrated jsonb null default '{}'::jsonb,
  started_at timestamp with time zone null default now(),
  completed_at timestamp with time zone null,
  updated_at timestamp with time zone null default now(),
  constraint user_progress_pkey primary key (id),
  constraint user_progress_user_id_module_id_key unique (user_id, module_id),
  constraint user_progress_module_id_fkey foreign KEY (module_id) references learning_modules (id) on delete CASCADE,
  constraint user_progress_user_id_fkey foreign KEY (user_id) references auth.users (id) on delete CASCADE,
  constraint user_progress_status_check check (
    (
      (status)::text = any (
        (
          array[
            'not_started'::character varying,
            'in_progress'::character varying,
            'completed'::character varying
          ]
        )::text[]
      )
    )
  )
) TABLESPACE pg_default;

create index IF not exists idx_user_progress_user on public.user_progress using btree (user_id, status) TABLESPACE pg_default;

create index IF not exists idx_user_progress_module on public.user_progress using btree (module_id, status) TABLESPACE pg_default;

create trigger update_user_progress_updated_at BEFORE
update on user_progress for EACH row
execute FUNCTION update_updated_at_column ();
